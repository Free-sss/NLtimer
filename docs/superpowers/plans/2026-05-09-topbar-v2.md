
在 `updateTimeLabelConfig` 方法之后添加：
```kotlin
    override fun getTopBarStyleFlow(): Flow<TopBarStyle> = dataStore.data.map { prefs ->
        val name = prefs[topBarStyleKey] ?: TopBarStyle.STANDARD.name
        try { TopBarStyle.valueOf(name) } catch (_: IllegalArgumentException) { TopBarStyle.STANDARD }
    }

    override suspend fun updateTopBarStyle(style: TopBarStyle) {
        dataStore.edit { prefs ->
            prefs[topBarStyleKey] = style.name
        }
    }
```

在 `companion object` 中添加 key：
```kotlin
        private val topBarStyleKey = stringPreferencesKey("top_bar_style")
```

将 RouteSettingsPopup 中 `onHomeLayoutChange` 和 `onShowTimeSideBarChange` 的回调改为引用上方的 `themeViewModel`：
```kotlin

                RouteSettingsPopup(
                    currentRoute = currentRoute,
                    onDismiss = { showSettingsPopup = false },
                    onHomeLayoutChange = { themeViewModel.onHomeLayoutChange(it) },
                    onShowTimeSideBarChange = { themeViewModel.onShowTimeSideBarToggle(it) }
                )
            }
```

- [ ] **步骤 3：修改 topBar lambda**

将 `topBar = { ... }` 替换为：
```kotlin
                topBar = {
                    if (topBarStyle == TopBarStyle.NONE) {
                        // 不渲染顶栏，内容扩展到状态栏下方
                    } else if (isSecondaryPage) {
                        TopAppBar(
                            title = { Text(topBarTitle) },
                            navigationIcon = {
                                IconButton(onClick = { navController.popBackStack() }) {
                                    Icon(
                                        imageVector = Icons.AutoMirrored.Filled.ArrowBack,
                                        contentDescription = "返回",
                                    )
                                }
                            },
                        )
                    } else {
                        AppTopAppBar(
                            title = topBarTitle,
                            onMenuClick = {
                                coroutineScope.launch { drawerState.open() }
                            },
                            onSettingClick = {
                                showSettingsPopup = true
                            },
                        )
                    }
                },
```

关键：TopBarStyle.NONE 时 topBar lambda 为空实现，Scaffold 不渲染任何 topBar 内容，内容区域自然扩展。
