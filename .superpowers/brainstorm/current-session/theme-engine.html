🎨 NLtimerTheme — 核心 Composable

// core:designsystem/NLtimerTheme.kt

@Composable
fun NLtimerTheme(
    theme: Theme = Theme(),
    content: @Composable () -> Unit
) {
    // 1. 确定 Dark/Light
    val isDark = when (theme.appTheme) {
        AppTheme.SYSTEM -> isSystemInDarkTheme()
        AppTheme.LIGHT -> false
        AppTheme.DARK -> true
    }

    // 2. 生成 ColorScheme
    val colorScheme = rememberDynamicColorScheme(
        primary = theme.seedColor,
        isDark = isDark,
        isAmoled = theme.isAmoled,
        style = theme.paletteStyle.toMPaletteStyle(),
    )

    // 3. 构建 Typography
    val typography = provideTypography(font = theme.font.toFontRes())

    // 4. 动画过渡（防闪烁）
    val animatedScheme = animateColorScheme(
        targetScheme = colorScheme,
        animationSpec = tween(durationMillis = 300)
    )

    // 5. 组装
    DynamicMaterialTheme(
        seedColor = theme.seedColor,
        isDark = isDark,
        isAmoled = theme.isAmoled,
        style = theme.paletteStyle.toMPaletteStyle(),
        typography = typography,
    ) {
        AnimatedContent(
            targetState = theme,
            transitionSpec = { fadeIn(tween(300)) + fadeOut(tween(300)) }
        ) {
            content()
        }
    }
}

@Composable
fun provideTypography(font: Int? = null): Typography {
    val selectedFont = font?.let { FontFamily(Font(font)) }
        ?: FontFamily.Default

    return Typography(
        displayLarge  = /* 默认值 */.copy(fontFamily = selectedFont),
        headlineLarge = /* 默认值 */.copy(fontFamily = selectedFont),
        titleLarge    = /* 默认值 */.copy(fontFamily = selectedFont),
        bodyLarge     = /* 默认值 */.copy(fontFamily = selectedFont),
        labelLarge    = /* 默认值 */.copy(fontFamily = selectedFont),
        // ... 全部 15 个 Material3 typography 层级
    )
}

🔀 枚举映射扩展

// core:designsystem/enumExt.kt

fun Fonts.toFontRes(): Int? = when (this) {
    Fonts.FIGTREE      -> R.font.figtree
    Fonts.GOOGLE_SANS  -> R.font.google_sans_flex
    Fonts.SYSTEM_DEFAULT -> null       // null → FontFamily.Default
}

fun Fonts.toDisplayString(): String = when (this) {
    Fonts.FIGTREE      -> "Figtree"
    Fonts.GOOGLE_SANS  -> "Google Sans"
    Fonts.SYSTEM_DEFAULT -> "System Default"
}

fun PaletteStyle.toMPaletteStyle(): com.materialkolor.PaletteStyle = when (this) {
    PaletteStyle.TONALSPOT   -> PaletteStyle.TonalSpot
    PaletteStyle.NEUTRAL     -> PaletteStyle.Neutral
    PaletteStyle.VIBRANT     -> PaletteStyle.Vibrant
    // ... 其余 6 种映射
}

fun AppTheme.toDisplayRes(): String = when (this) {
    AppTheme.LIGHT  -> "Light"
    AppTheme.DARK   -> "Dark"
    AppTheme.SYSTEM -> "Follow System"
}
