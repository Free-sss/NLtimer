// core:designsystem/NLtimerTheme.kt — 修正版

@OptIn(ExperimentalMaterial3ExpressiveApi::class)
@Composable
fun NLtimerTheme(
    theme: Theme = Theme(),
    content: @Composable () -> Unit
) {
    // 确定 Dark/Light
    val isDark = when (theme.appTheme) {
        AppTheme.SYSTEM -> isSystemInDarkTheme()
        AppTheme.LIGHT -> false
        AppTheme.DARK -> true
    }

    // 构建 Typography（普通函数，非 Composable）
    val typography = provideTypography(font = theme.font.toFontRes())

    // DynamicMaterialTheme 内部自行生成 ColorScheme
    // 无需手动 rememberDynamicColorScheme + animateColorScheme
    DynamicMaterialTheme(
        seedColor = theme.seedColor,
        isDark = isDark,
        isAmoled = theme.isAmoled,
        style = theme.paletteStyle.toMPaletteStyle(),
        typography = typography,
    ) {
        // AnimatedContent 靠淡入淡出掩盖主题切换
        // 仅对引起强烈视觉变化的属性做过渡
        AnimatedContent(
            targetState = theme.appTheme to theme.isAmoled,
            transitionSpec = { fadeIn(tween(300)) + fadeOut(tween(300)) },
            label = "theme-transition"
        ) {
            content()
        }
    }
}

⚠️ 修正 3：provideTypography 降级为普通函数

// 不再标记 @Composable，避免误导和多余重组开销

fun provideTypography(font: Int? = null): Typography {
    val selectedFont = font?.let { FontFamily(Font(it)) }
        ?: FontFamily.Default

    return Typography(
        displayLarge  = TYPOGRAPHY.displayLarge.copy(fontFamily = selectedFont),
        headlineLarge = TYPOGRAPHY.headlineLarge.copy(fontFamily = selectedFont),
        titleLarge    = TYPOGRAPHY.titleLarge.copy(fontFamily = selectedFont),
        // ... 全部 15 个层级
    )
}

private val TYPOGRAPHY = Typography()  // 静态单例，避免重复创建
