设计 §1：ExpressivenessPreset 架构

新增"表达力预设"作为主旋钮，联动所有视觉维度

核心理念

ExpressivenessPreset 是一个"主预设"——切换它时自动设置下面所有子维度的推荐值。
高级用户仍可单独微调每个子维度（就像现在 cornerPreset/borderPreset 一样）。
一键切换氛围，精细控制不丢。

数据模型变更

// 新增枚举
enum class ExpressivenessPreset {
    SUBDUED, STANDARD, EXPRESSIVE
}

// 新增卡片色策略
enum class CardColorStrategy {
    SURFACE,              // 纯 surfaceContainer 灰底
    TINTED_PRIMARY,       // primaryContainer 浅染
    MULTI_CONTAINER       // primary/secondary/tertiaryContainer 拼盘
}

// StyleConfig 扩展
data class StyleConfig(
    val cornerPreset: CornerPreset = STANDARD,
    val borderPreset: BorderPreset = STANDARD,
    val alphaPreset: AlphaPreset = STANDARD,
    val expressiveness: ExpressivenessPreset = SUBDUED,  // 新
    val cardColorStrategy: CardColorStrategy = SURFACE,   // 新
    val iconContainerSize: IconContainerSize = NONE,      // 新
    val timerTypography: TimerTypography = HEADLINE,       // 新
    val pressedShape: PressedShapeLevel = OFF,            // 新
    val wavyProgress: WavyProgressLevel = OFF,            // 新
)

// 一键设置函数
fun ExpressivenessPreset.toStyleConfig(): StyleConfig = when (this) {
    SUBDUED    -> StyleConfig(COMPACT, STANDARD, SUBTLE, ...)
    STANDARD   -> StyleConfig(STANDARD, THIN, STANDARD, ...)
    EXPRESSIVE -> StyleConfig(SOFT, NONE, VIVID, ...)
}
